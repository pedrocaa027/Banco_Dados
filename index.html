create table if not exists colaborador(

colaborador_nome varchar (45) not full,

colaborador_cpf varchar (45) not full,

colaborador_endereco int,

colaborador_matricula varchar (45) not full,

colaborador_telefone varchar (45) not full );



creatre table if not exists fornecedor(

forncecedor_nome varchar(45) not full,

fornecedor_endereço int,

fornecedor_placa int,

fornecedor_telefone varchar (45) not full,

fornecedor_data date not full );



create table if not exists produto(

produto_validade date not full,

produto_nome varchar (45),

produto_cod int,

produto_quantidade int );



create table if not exists cliente (

cliente_nome varchar(45),

cliente_telefone varchar(45),

cliente_endereço int );



create table if not exists pedidos(

pedidos_endereco int,

pedidos_data date not full,

pedidos_valor int );
